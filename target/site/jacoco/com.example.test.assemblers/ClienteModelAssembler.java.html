package com.example.test.assemblers;

import org.springframework.hateoas.EntityModel;
import org.springframework.hateoas.server.RepresentationModelAssembler;
import org.springframework.stereotype.Component;
import static org.springframework.hateoas.server.mvc.WebMvcLinkBuilder.*;
import com.example.test.controller.ClienteControllerV2;
import com.example.test.model.Cliente;
import org.springframework.lang.NonNull;

@Component
public class ClienteModelAssembler implements RepresentationModelAssembler<Cliente, EntityModel<Cliente>>{
    
    @Override
    public EntityModel<Cliente> toModel(@NonNull Cliente cliente){

        return EntityModel.of(cliente,
        linkTo(methodOn(ClienteControllerV2.class).getClienteByCodigo(cliente.getRut())).withSelfRel(),
        linkTo(methodOn(ClienteControllerV2.class).getAllCliente()).withRel("Clientes"));
    }
}
