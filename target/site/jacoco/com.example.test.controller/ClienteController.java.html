package com.example.test.controller;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.DeleteMapping;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import com.example.test.model.Cliente;
import com.example.test.services.ClienteService;

import io.swagger.v3.oas.annotations.Operation;
import io.swagger.v3.oas.annotations.tags.Tag;

@RestController
@RequestMapping("/cotizacion/cliente")
@Tag(name= "Cliente", description = "Operaciones relacionadas con los clientes")
public class ClienteController {
    @Autowired
    private ClienteService clienteService;

    @GetMapping
    @Operation(summary = "Obtener todos los clientes", description = "Obtiene una lista de todos los clientes")
    public ResponseEntity<List<Cliente>> obtenerClientes(){
        List<Cliente> clientes = clienteService.getClientes();
        if(clientes.isEmpty()){
            return ResponseEntity.noContent().build(); 
        }
        return ResponseEntity.ok(clientes);
    }

    @GetMapping("{rut}")
    @Operation(summary = "Obtener un cliete por su rut", description = "Busca un cliente por su rut")
    public ResponseEntity<Cliente> buscarClientePorRut(@PathVariable Long rut){
        try{
            Cliente cliente = clienteService.obtenerClientePorRut(rut);
            return ResponseEntity.ok(cliente);
        }catch (Exception e){
            return ResponseEntity.notFound().build();
        }
    }

    @PostMapping
    @Operation(summary = "Crea un nuevo cliente", description = "Agrega un nuevo cliente")
    public ResponseEntity<Cliente> saveCliente(@RequestBody Cliente cliente){
        Cliente clieteNuevo = clienteService.agregarCliente(cliente);
        return ResponseEntity.status(HttpStatus.CREATED).body(clieteNuevo);
    }

    @PutMapping("/act/{id}")
    @Operation(summary = "Actualizar un cliente", description = "Actualiza un cliente por su rut")
    public ResponseEntity<Cliente> actualizarCliente(@PathVariable Long id,@RequestBody Cliente cliente){
        try{
            Cliente cli = clienteService.obtenerClientePorRut(id);
            cli.setRut(id);
            cli.setDv(cliente.getDv());
            cli.setNombre(cliente.getNombre());
            cli.setTelefono(cliente.getTelefono());
            cli.setCorreo(cliente.getCorreo());
            cli.setNombre_usuario(cliente.getNombre_usuario());
            cli.setContrasena(cliente.getContrasena());

            clienteService.modificarCliente(cli);
            return ResponseEntity.ok(cliente);
        } catch (Exception e){
            return ResponseEntity.notFound().build();
        } 
    }

    @DeleteMapping("{rut}")
    @Operation(summary = "Elimina un cliente", description = "Elimina un cliente por su rut")
    public ResponseEntity<?> eliminarCliente(@PathVariable Long rut){
        try{
            clienteService.eliminarCliente(rut);
            return ResponseEntity.noContent().build();
        } catch (Exception e){
            return ResponseEntity.notFound().build();
        }

    }
}
