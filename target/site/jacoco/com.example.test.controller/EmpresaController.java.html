package com.example.test.controller;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.DeleteMapping;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;


import com.example.test.model.Empresa;
import com.example.test.services.EmpresaService;

import io.swagger.v3.oas.annotations.Operation;
import io.swagger.v3.oas.annotations.tags.Tag;

@RestController
@RequestMapping("/cotizacion/empresa")
@Tag(name = "Empresa",description = "Operaciones relacionadas con las empresas")
public class EmpresaController {

    @Autowired
    public EmpresaService empresaService;

    @GetMapping
    @Operation(summary = "Obtener todos las empresas", description = "Obtiene una lista de todos las empresas")
    public ResponseEntity<List<Empresa>> obtenerEmpresas(){
       List<Empresa> empresas = empresaService.getEmpresas();
        if(empresas.isEmpty()){
            return ResponseEntity.noContent().build(); 
        }
        return ResponseEntity.ok(empresas);
    }

    @PostMapping
    @Operation(summary = "Crea una nueva empresa", description = "Agrega una nueva empresa")
    public ResponseEntity<Empresa> saveEmpresa(@RequestBody Empresa empresa){
        Empresa empresaNuevo = empresaService.agregarEmpresa(empresa);
        return ResponseEntity.status(HttpStatus.CREATED).body(empresaNuevo);
    }


    @GetMapping("{rut}")
    @Operation(summary = "Obtener un empresa por su rut", description = "Busca un empresa por su rut")
    public ResponseEntity<Empresa> buscarEmpresaPorRut(@PathVariable Long rut){
        try{
            Empresa empresa = empresaService.buscarEmpresa(rut);
            return ResponseEntity.ok(empresa);
        }catch (Exception e){
            return ResponseEntity.notFound().build();
        }
    }

    @PutMapping("/act/{id}")
    @Operation(summary = "Actualizar una empresa", description = "Actualiza una empresa por su rut")
    public ResponseEntity<Empresa> actualizarEmpresa(@PathVariable Long id,@RequestBody Empresa empresa){
        try{
            Empresa emp = empresaService.buscarEmpresa(id);
            emp.setRut(id);
            emp.setDv(empresa.getDv());
            emp.setNombre(empresa.getNombre());
            emp.setTelefono(empresa.getTelefono());
            emp.setCorreo(empresa.getCorreo());
            emp.setNombre_representante(empresa.getNombre_representante());
            emp.setRol(empresa.getRol());

            empresaService.modificarEmpresa(emp);
            return ResponseEntity.ok(empresa);
        } catch (Exception e){
            return ResponseEntity.notFound().build();
        } 
    }

    @DeleteMapping("{rut}")
    @Operation(summary = "Elimina una empresa", description = "Elimina una empresa por su rut")
    public ResponseEntity<?> eliminarEmpresa(@PathVariable Long rut){
        try{
            empresaService.eliminarEmpresa(rut);
            return ResponseEntity.noContent().build();
        } catch (Exception e){
            return ResponseEntity.notFound().build();
        }

    }
}
