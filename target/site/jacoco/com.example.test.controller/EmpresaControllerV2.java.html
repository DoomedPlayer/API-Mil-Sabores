package com.example.test.controller;

import static org.springframework.hateoas.server.mvc.WebMvcLinkBuilder.linkTo;
import static org.springframework.hateoas.server.mvc.WebMvcLinkBuilder.methodOn;

import java.util.List;
import java.util.stream.Collectors;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.hateoas.CollectionModel;
import org.springframework.hateoas.EntityModel;
import org.springframework.hateoas.MediaTypes;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.DeleteMapping;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import com.example.test.assemblers.EmpresaModelAssembler;
import com.example.test.model.Empresa;
import com.example.test.services.EmpresaService;


@RestController
@RequestMapping("/empresas")
public class EmpresaControllerV2 {
@Autowired
    public EmpresaService empresaService;

    @Autowired
    private EmpresaModelAssembler assembler;

    @GetMapping(produces = MediaTypes.HAL_JSON_VALUE)
    public CollectionModel<EntityModel<Empresa>> getAllEmpresas(){
        List<EntityModel<Empresa>> empresas= empresaService.getEmpresas().stream()
        .map(assembler::toModel).collect(Collectors.toList());
        return CollectionModel.of(empresas,
                linkTo(methodOn(EmpresaControllerV2.class).getAllEmpresas()).withSelfRel());
    }

    @PostMapping(produces = MediaTypes.HAL_JSON_VALUE)
    public ResponseEntity<EntityModel<Empresa>> agregarEmpresa(@RequestBody Empresa empresa){
        Empresa newEmpresa= empresaService.agregarEmpresa(empresa);
        return ResponseEntity.created(linkTo(methodOn(EmpresaControllerV2.class).getEmpresaByCodigo(newEmpresa.getRut())).toUri())
        .body(assembler.toModel(newEmpresa));
    }

    @GetMapping(value = "/{id}",produces = MediaTypes.HAL_JSON_VALUE)
    public EntityModel<Empresa> getEmpresaByCodigo(@PathVariable long id){
        Empresa empresa= empresaService.buscarEmpresa(id);
        return assembler.toModel(empresa);
    }

    @PutMapping(value = "/act/{id}",produces = MediaTypes.HAL_JSON_VALUE)
    public ResponseEntity<EntityModel<Empresa>> modificarEmpresa(@PathVariable long id,@RequestBody Empresa empresa){
   Empresa existingEmpresa;
        try {
            existingEmpresa = empresaService.buscarEmpresa(id);
        } catch (java.util.NoSuchElementException e) {
            return ResponseEntity.notFound().build();
        }

        existingEmpresa.setDv(empresa.getDv());
        existingEmpresa.setNombre(empresa.getNombre());
        existingEmpresa.setTelefono(empresa.getTelefono());
        existingEmpresa.setCorreo(empresa.getCorreo());
        existingEmpresa.setNombre_representante(empresa.getNombre_representante());
        existingEmpresa.setRol(empresa.getRol());

        Empresa updatedEmpresa = empresaService.modificarEmpresa(existingEmpresa);
        return ResponseEntity.ok(assembler.toModel(updatedEmpresa));
    }

    @DeleteMapping(value = "{id}",produces = MediaTypes.HAL_JSON_VALUE)
    public ResponseEntity<?> eliminarEmpresa(@PathVariable long id){
            empresaService.eliminarEmpresa(id);
            return ResponseEntity.noContent().build();
    }


}
