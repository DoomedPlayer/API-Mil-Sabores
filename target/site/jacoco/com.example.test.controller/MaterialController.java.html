package com.example.test.controller;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.DeleteMapping;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import com.example.test.model.Material;
import com.example.test.services.MaterialService;

import io.swagger.v3.oas.annotations.Operation;
import io.swagger.v3.oas.annotations.tags.Tag;

@RestController
@RequestMapping("/cotizacion/material")
@Tag(name = "Material", description = "Operaciones relacionadas con los materiales")
public class MaterialController {
    @Autowired
    public MaterialService materialService;

    @GetMapping
    @Operation(summary = "Obtener todos los materiales", description = "Obtiene una lista de todos los materiales")
    public ResponseEntity<List<Material>> obtenerLista(){
        List<Material> materiales= materialService.obtenerLista();
        if(materiales.isEmpty()){
            return ResponseEntity.noContent().build();
        }
        return ResponseEntity.ok(materiales);
    }

    @PostMapping
    @Operation(summary = "Crea un nuevo material", description = "Agrega un nuevo material")
    public ResponseEntity<Material> agregarMat(@RequestBody Material material){
        Material materialNuevo = materialService.agregarMat(material);
        return ResponseEntity.status(HttpStatus.CREATED).body(materialNuevo);
    }

    @GetMapping("{id}")
    @Operation(summary = "Obtener un material por su id", description = "Busca un material por su id")
    public ResponseEntity<Material> buscarMat(@PathVariable int id){
        try{
            Material material= materialService.buscarMat(id);
            return ResponseEntity.ok(material);
        }catch (Exception e){
            return ResponseEntity.notFound().build();
        }
    }

    @PutMapping("/act/{id}")
    @Operation(summary = "Actualizar un material", description = "Actualiza un material por su id")
    public ResponseEntity<Material> actualizarMat(@PathVariable int id,@RequestBody Material material){
        try{
            Material mat= materialService.buscarMat(id);
            mat.setIdMaterial(id);
            mat.setNombre(material.getNombre());
            mat.setLargo(material.getLargo());
            mat.setAncho(material.getAncho());
            mat.setCostoUnitario(material.getCostoUnitario());
            materialService.actualizarMat(mat);
            return ResponseEntity.ok(material);
        }catch(Exception e){
            return ResponseEntity.notFound().build();
        }
    }

    @DeleteMapping("{id}")
    @Operation(summary = "Elimina un material", description = "Elimina un material por su id")
    public ResponseEntity<?> eliminarMat(@PathVariable int id){
        try{
            materialService.eliminarMat(id);
            return ResponseEntity.noContent().build();
        }catch(Exception e){
            return ResponseEntity.notFound().build();
        }
    }
}
