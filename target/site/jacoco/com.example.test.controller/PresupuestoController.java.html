package com.example.test.controller;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.DeleteMapping;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import com.example.test.model.Presupuesto;
import com.example.test.services.PresupuestoService;

import io.swagger.v3.oas.annotations.Operation;
import io.swagger.v3.oas.annotations.tags.Tag;


@RestController
@RequestMapping("/cotizacion/presupuesto")
@Tag(name = "Presupuesto", description = "Operaciones relacionadas con los presupuestos")
public class PresupuestoController {
    @Autowired
    public PresupuestoService presService;

    @GetMapping
    @Operation(summary = "Obtener todos los presupuestos", description = "Obtiene una lista de todos los presupuestos")
    public ResponseEntity<List<Presupuesto>> obtenerPres(){
        List<Presupuesto> presupuestos = presService.obtenerPres();
        if(presupuestos.isEmpty()){
            return ResponseEntity.noContent().build();
        }
        return ResponseEntity.ok(presupuestos);
    }

    @PostMapping
    @Operation(summary = "Crea un nuevo presupuesto", description = "Agrega un nuevo presupuesto")
    public ResponseEntity<Presupuesto> agregarPres(@RequestBody Presupuesto presupuesto){
        Presupuesto presuNuevo = presService.agregarPres(presupuesto);
        return ResponseEntity.status(HttpStatus.CREATED).body(presuNuevo);
    }

    @GetMapping("{id}")
    @Operation(summary = "Obtener un presupuesto por su id", description = "Busca un presupuesto por su id")
    public ResponseEntity<Presupuesto> buscarPres(@PathVariable int id){
        try{
            Presupuesto presupuesto = presService.buscarPres(id);
            return ResponseEntity.ok(presupuesto);
        }catch (Exception e){
            return ResponseEntity.notFound().build();
        }
    }

    @PutMapping("/act/{id}")
    @Operation(summary = "Actualizar un presupuesto", description = "Actualiza un presupuesto por su id")
    public ResponseEntity<Presupuesto> actualizarPres(@PathVariable int id,@RequestBody Presupuesto presupuesto){
        try{
            Presupuesto pre = presService.buscarPres(id);
            pre.setIdPresupueto(id);
            pre.setCostoManoObra(presupuesto.getCostoManoObra());
            pre.setCostoMaterial(presupuesto.getCostoMaterial());
            pre.setCostoOperacion(presupuesto.getCostoOperacion());
            pre.setTotalPresupuesto(presupuesto.getTotalPresupuesto());
            pre.setUtilidadEstimada(presupuesto.getUtilidadEstimada());
            presService.actualizarPres(pre);
            return ResponseEntity.ok(presupuesto);
        } catch (Exception e){
            return ResponseEntity.notFound().build();
        }
    }

    @DeleteMapping("{id}")
    @Operation(summary = "Elimina un presupuesto", description = "Elimina un presupuesto por su id")
    public ResponseEntity<?> eliminarPres(@PathVariable int id){
        try{
            presService.eliminarPres(id);
            return ResponseEntity.noContent().build();
        } catch (Exception e){
            return ResponseEntity.notFound().build();
        }
    }
}
