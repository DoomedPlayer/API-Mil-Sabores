package com.example.test.controller;

import static org.springframework.hateoas.server.mvc.WebMvcLinkBuilder.linkTo;
import static org.springframework.hateoas.server.mvc.WebMvcLinkBuilder.methodOn;

import java.util.List;
import java.util.stream.Collectors;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.hateoas.CollectionModel;
import org.springframework.hateoas.EntityModel;
import org.springframework.hateoas.MediaTypes;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.DeleteMapping;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import com.example.test.assemblers.PresupuestoModelAssembler;
import com.example.test.model.Presupuesto;
import com.example.test.services.PresupuestoService;


@RestController
@RequestMapping("/presupuesto")
public class PresupuestoControllerV2 {

    @Autowired
    public PresupuestoService presupuestoService;

    @Autowired
    private PresupuestoModelAssembler assembler;

    @GetMapping(produces = MediaTypes.HAL_JSON_VALUE)
    public CollectionModel<EntityModel<Presupuesto>> getAllPresupuestos(){
        List<EntityModel<Presupuesto>> presupuesto= presupuestoService.obtenerPres().stream()
        .map(assembler::toModel).collect(Collectors.toList());
        return CollectionModel.of(presupuesto,
                linkTo(methodOn(PresupuestoControllerV2.class).getAllPresupuestos()).withSelfRel());
    }

    @PostMapping(produces = MediaTypes.HAL_JSON_VALUE)
    public ResponseEntity<EntityModel<Presupuesto>> createCot(@RequestBody Presupuesto presupuesto){
        Presupuesto newPresupuesto= presupuestoService.agregarPres(presupuesto);
        return ResponseEntity.created(linkTo(methodOn(PresupuestoControllerV2.class).getPresupuestoByCodigo(newPresupuesto.getIdPresupueto())).toUri())
        .body(assembler.toModel(newPresupuesto));
    }

    @GetMapping(value = "/{id}",produces = MediaTypes.HAL_JSON_VALUE)
    public EntityModel<Presupuesto> getPresupuestoByCodigo(@PathVariable int id){
        Presupuesto presupuesto= presupuestoService.buscarPres(id);
        return assembler.toModel(presupuesto);
    }

    @PutMapping(value = "/act/{id}",produces = MediaTypes.HAL_JSON_VALUE)
    public ResponseEntity<EntityModel<Presupuesto>> actualizarCot(@PathVariable int id,@RequestBody Presupuesto presupuesto){
            presupuesto.setIdPresupueto(id);
            Presupuesto updatePresupuesto = presupuestoService.actualizarPres(presupuesto);
            return ResponseEntity.ok(assembler.toModel(updatePresupuesto));
    }

    @DeleteMapping(value = "{id}",produces = MediaTypes.HAL_JSON_VALUE)
    public ResponseEntity<?> eliminarCot(@PathVariable int id){
            presupuestoService.eliminarPres(id);
            return ResponseEntity.noContent().build();
    }

}
