package com.example.test.services;

import java.util.List;
import java.util.Optional;

import org.hibernate.Hibernate;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.example.test.model.Cliente;
import com.example.test.model.Cotizacion;
import com.example.test.model.DetPedido;
import com.example.test.model.Pedido;
import com.example.test.repository.CotizacionRepositoryI;
import com.example.test.repository.PedidoRepositoryI;

import jakarta.transaction.Transactional;

@Service
@Transactional
public class CotizacionService {
    @Autowired
    private CotizacionRepositoryI cRepository;

        @Autowired
    private PedidoRepositoryI pedidoRepository;

    public List<Cotizacion> obtenerCotizacion(){
        return cRepository.findAll();
    }

    public Cotizacion agregarCot(Cotizacion cotizacion){
       if (cotizacion.getPedido() != null && cotizacion.getPedido().getIdPedido() != null) {
            Optional<Pedido> pedidoExistente = pedidoRepository.findById(cotizacion.getPedido().getIdPedido());

            if (pedidoExistente.isPresent()) {
                cotizacion.setPedido(pedidoExistente.get());
            } else {
                throw new RuntimeException("Pedido con ID " + cotizacion.getPedido().getIdPedido() + " no encontrado para la cotización.");
            }
        } else {
             throw new RuntimeException("Información del Pedido (ID) es requerida para la cotización.");
        }
        Cotizacion cotizacionGuardada = cRepository.save(cotizacion);
        if (cotizacionGuardada.getPedido() != null) {
            Hibernate.initialize(cotizacionGuardada.getPedido());
            if (cotizacionGuardada.getPedido().getPersona() != null) {
                 Hibernate.initialize(cotizacionGuardada.getPedido().getPersona());
                 if (cotizacionGuardada.getPedido().getPersona() instanceof Cliente) {
                     Hibernate.initialize(((Cliente) cotizacionGuardada.getPedido().getPersona()).getNombre_usuario());
                     Hibernate.initialize(((Cliente) cotizacionGuardada.getPedido().getPersona()).getContrasena());
                 }
            }
            if (cotizacionGuardada.getPedido().getListaProductos() != null) {
                Hibernate.initialize(cotizacionGuardada.getPedido().getListaProductos());
                for (DetPedido det : cotizacionGuardada.getPedido().getListaProductos()) {
                     if (det.getProducto() != null) {
                        Hibernate.initialize(det.getProducto());
                     }
                }
            }
        }
        return cotizacionGuardada;
    }

    public Cotizacion buscarCot(int id){
        return cRepository.findById(id).get();
    }

    public Cotizacion actualizarCot(Cotizacion cotizacion){
        return cRepository.save(cotizacion);
    }

    public void eliminarCot(int id){
        cRepository.deleteById(id);
    }
}
