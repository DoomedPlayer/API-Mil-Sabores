package com.example.test.services;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.example.test.model.Material;
import com.example.test.repository.MaterialRepositoryI;

import jakarta.transaction.Transactional;

@Service
@Transactional
public class MaterialService {
    @Autowired
    public MaterialRepositoryI materialRepository;

    public List<Material> obtenerLista(){
        return materialRepository.findAll();
    }

    public Material buscarMat(int id){
        return materialRepository.findById(id).get();
    }

    public Material agregarMat(Material material){
        return materialRepository.save(material);
    }

    public Material actualizarMat(Material material){
        return materialRepository.save(material);
    }

    public void eliminarMat(int id){
        materialRepository.deleteById(id);
    }
}
