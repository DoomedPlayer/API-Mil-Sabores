package com.example.test.services;

import java.util.ArrayList;
import java.util.List;
import java.util.Optional;

import org.hibernate.Hibernate;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.example.test.model.Cliente;
import com.example.test.model.DetPedido;
import com.example.test.model.Pedido;
import com.example.test.model.Producto;
import com.example.test.repository.ClienteRepositoryI;
import com.example.test.repository.PedidoRepositoryI;
import com.example.test.repository.ProductoRepositoryI;

import jakarta.transaction.Transactional;

@Service
@Transactional
public class PedidoService {
    @Autowired
    public PedidoRepositoryI pedidoRepository;

    @Autowired
    private ClienteRepositoryI clienteRepository;

    @Autowired
    private ProductoRepositoryI productoRepository;

    public List<Pedido> obtenerPedidos(){
        return pedidoRepository.findAll();
    }

    public Pedido buscarPedido(Long id){
        return pedidoRepository.findById(id).get();
    }

    public Pedido agregarPedido(Pedido pedido){
        if (pedido.getPersona() != null && pedido.getPersona().getRut() != null) {
            Optional<Cliente> clienteExistente = clienteRepository.findById(pedido.getPersona().getRut());

            if (clienteExistente.isPresent()) {
                pedido.setPersona(clienteExistente.get());
            } else {
                throw new RuntimeException("Cliente con RUT " + pedido.getPersona().getRut() + " no encontrado.");
            }
        } else {

             throw new RuntimeException("Informaci贸n de Persona (Cliente) con RUT es requerida.");
        }
        if (pedido.getListaProductos() != null) {
            List<DetPedido> detallesParaGuardar = new ArrayList<>(); 
            for (DetPedido det : pedido.getListaProductos()) {
                det.setPedido(pedido);
                if (det.getProducto() != null && det.getProducto().getId() != 0) {
                     Optional<Producto> productoExistente = productoRepository.findById(det.getProducto().getId());
                     if (productoExistente.isPresent()) {
                         det.setProducto(productoExistente.get()); 
                     } else {
                         throw new RuntimeException("Producto con ID " + det.getProducto().getId() + " no encontrado para el detalle del pedido.");
                     }
                 } else {
                     throw new RuntimeException("Informaci贸n de Producto con ID es requerida en Detalle de Pedido.");
                 }

                detallesParaGuardar.add(det);
            }
            pedido.setListaProductos(detallesParaGuardar);

        } else {
             pedido.setListaProductos(new ArrayList<>());
        }

        Pedido pedidoGuardado = pedidoRepository.save(pedido);
        if (pedidoGuardado.getListaProductos() != null) {
            for (DetPedido det : pedidoGuardado.getListaProductos()) {
                 Hibernate.initialize(det.getProducto()); // Inicializa el Producto dentro de cada DetPedido
            }
        }
        return pedidoGuardado;
    }

    public Pedido actualizarPedido(Pedido pedido){
        Optional<Pedido> pedidoExistenteOpt = pedidoRepository.findById(pedido.getIdPedido());
        if (pedidoExistenteOpt.isPresent()) {
            Pedido pedidoExistente = pedidoExistenteOpt.get();
            pedidoExistente.setFechaEntrega(pedido.getFechaEntrega());
            pedidoExistente.setFechaPedido(pedido.getFechaPedido());
             if (pedido.getPersona() != null && pedido.getPersona().getRut() != null) {
                Optional<Cliente> clienteExistente = clienteRepository.findById(pedido.getPersona().getRut());
                if (clienteExistente.isPresent()) {
                    pedidoExistente.setPersona(clienteExistente.get());
                } else {
                    throw new RuntimeException("Cliente con RUT " + pedido.getPersona().getRut() + " no encontrado para actualizar.");
                }
            } else {
                 throw new RuntimeException("Informaci贸n de Persona (Cliente) es requerida para actualizar.");
            }
            if (pedido.getListaProductos() != null) {
                 List<DetPedido> nuevosDetalles = new ArrayList<>();
                 for (DetPedido det : pedido.getListaProductos()) {
                     det.setPedido(pedidoExistente);
                     if (det.getProducto() != null && det.getProducto().getId() != 0) {
                         Optional<Producto> productoExistente = productoRepository.findById(det.getProducto().getId());
                         if (productoExistente.isPresent()) {
                             det.setProducto(productoExistente.get());
                         } else {
                             throw new RuntimeException("Producto con ID " + det.getProducto().getId() + " no encontrado para el detalle del pedido al actualizar.");
                         }
                     } else {
                         throw new RuntimeException("Informaci贸n de Producto con ID es requerida en Detalle de Pedido al actualizar.");
                     }

                     nuevosDetalles.add(det);
                 }
                 pedidoExistente.getListaProductos().clear(); 
                 pedidoExistente.getListaProductos().addAll(nuevosDetalles);
            } else {
                pedidoExistente.getListaProductos().clear();
            }

            Pedido pedidoActualizado = pedidoRepository.save(pedidoExistente);

             if (pedidoActualizado.getListaProductos() != null) {
                Hibernate.initialize(pedidoActualizado.getListaProductos());
                 for (DetPedido det : pedidoActualizado.getListaProductos()) {
                     Hibernate.initialize(det.getProducto()); 
                 }
             }

            return pedidoActualizado;

        } else {
            throw new RuntimeException("Pedido con ID " + pedido.getIdPedido() + " no encontrado para actualizar.");
        }
    }

    public void eliminarPedido(Long id){
        pedidoRepository.deleteById(id);
    }
}
