package com.example.test.services;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.example.test.model.Presupuesto;
import com.example.test.repository.PresupuestoRepositoryI;

import jakarta.transaction.Transactional;

@Service
@Transactional
public class PresupuestoService {
    @Autowired
    public PresupuestoRepositoryI presupuestorRep;

    public List<Presupuesto> obtenerPres(){
        return presupuestorRep.findAll();
    }

    public Presupuesto buscarPres(int id){
        return presupuestorRep.findById(id).get();
    }

    public Presupuesto agregarPres(Presupuesto presupuesto){
        return presupuestorRep.save(presupuesto);
    }

    public Presupuesto actualizarPres(Presupuesto presupuesto){
        return presupuestorRep.save(presupuesto);
    }

    public void eliminarPres(int id){
        presupuestorRep.deleteById(id);
    }
}
